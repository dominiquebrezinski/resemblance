  def self.generate_n_set(text, options = {})
    prefs = {:evaluation_method => :ngraph}.merge(options)
    case prefs[:evaluation_method]
      when :ngraph, 'ngraph'
        Resemblance::NGraphSet.new(text, options)
      when :ngram, 'ngram'
        Resemblance::NGramSet.new(text, options)
      else
        Resemblance::NGraphSet.new(text, options)
    end
  end
