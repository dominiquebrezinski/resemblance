    def min_and_max_r
      largest_r_cluster = @centroids.index(@centroids.max {|a,b| a.last <=> b.last})
      smallest_r_cluster = @centroids.index(@centroids.min {|a,b| a.last <=> b.last})
      [@centroids[smallest_r_cluster].last, @centroids[largest_r_cluster].last]
    end
